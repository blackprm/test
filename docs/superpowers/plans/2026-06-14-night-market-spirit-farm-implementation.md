```ts
import { MUTATIONS, SEEDS } from "./NightMarketConfig";
import { MutationId, NightMarketOrder, PlotState, SeedId } from "./NightMarketTypes";

export function nowSeconds(): number {
    return Math.floor(Date.now() / 1000);
}

export function getSeed(seedId: SeedId) {
    return SEEDS.find(seed => seed.id === seedId);
}

export function getMutation(mutationId: MutationId) {
    return MUTATIONS.find(mutation => mutation.id === mutationId) || MUTATIONS[0];
}

export function isPlotMature(plot: PlotState, atSeconds: number): boolean {
    if (!plot.seedId) return false;
    const seed = getSeed(plot.seedId as SeedId);
    return !!seed && atSeconds - plot.plantedAt >= seed.growSeconds;
}

export function rollMutation(seed: SeedId, salt: number): MutationId {
    const roll = Math.abs(Math.sin((salt + seed.length) * 12.9898) * 43758.5453) % 1;
    if (roll > 0.985) return "wild";
    if (roll > 0.955) return "giant";
    if (roll > 0.9) return "lucky";
    if (roll > 0.78) return "glow";
    return "none";
}

export function cropValue(seedId: SeedId, mutationId: MutationId, order: NightMarketOrder | null): number {
    const seed = getSeed(seedId);
    if (!seed) return 0;
    const mutation = getMutation(mutationId);
    const orderMultiplier = order && order.seedId === seedId && order.mutationId === mutationId ? order.multiplier : 1;
    return Math.floor(seed.baseValue * mutation.valueMultiplier * orderMultiplier);
}

export function stealReward(seedId: SeedId, mutationId: MutationId, order: NightMarketOrder | null): number {
    return Math.max(8, Math.floor(cropValue(seedId, mutationId, order) * 0.35));
}

export function upgradeCost(baseLevel: number): number {
    return 180 + baseLevel * baseLevel * 120;
}
```

Create `PlayerFarmModuleS.ts`:

```ts
import { ORDER_SECONDS, SEEDS, SHIELD_SECONDS, STEAL_COOLDOWN_SECONDS } from "./NightMarketConfig";
import { cropValue, isPlotMature, nowSeconds, rollMutation, stealReward } from "./NightMarketMath";
import { PlayerFarmData } from "./PlayerFarmData";
import { FarmSnapshot, MutationId, NightMarketOrder, SeedId } from "./NightMarketTypes";
import { PlayerFarmModuleC } from "./PlayerFarmModuleC";

export class PlayerFarmModuleS extends ModuleS<PlayerFarmModuleC, PlayerFarmData> {
    private _order: NightMarketOrder = {
        seedId: "carrot",
        mutationId: "none",
        multiplier: 2,
        expiresAt: 0,
    };

    protected onStart(): void {
        this.refreshOrderIfNeeded();
    }

    protected onPlayerEnterGame(player: Player): void {
        this.refreshOrderIfNeeded();
        this.getClient(player).net_receiveSnapshot(this.snapshotFor(player));
    }

    public net_requestSnapshot(): void {
        this.refreshOrderIfNeeded();
        this.getClient(this.currentPlayer).net_receiveSnapshot(this.snapshotFor(this.currentPlayer));
    }

    public net_plant(plotIndex: number, seedId: SeedId): void {
        this.refreshOrderIfNeeded();
        const data = this.currentData;
        const seed = SEEDS.find(item => item.id === seedId);
        if (!seed || data.unlockedSeeds.indexOf(seedId) < 0) {
            this.reply(false, "种子未解锁");
            return;
        }
        const plot = data.plots[plotIndex];
        if (!plot || plot.seedId) {
            this.reply(false, "这块田不能种");
            return;
        }
        if (data.gold < seed.price) {
            this.reply(false, "金币不足");
            return;
        }
        data.gold -= seed.price;
        plot.seedId = seedId;
        plot.plantedAt = nowSeconds();
        plot.mutationId = rollMutation(seedId, plot.plantedAt + this.currentPlayer.playerId);
        plot.stolenCooldownUntil = 0;
        data.save(true);
        this.reply(true, "种下了 " + seed.name);
    }

    public net_harvest(plotIndex: number): void {
        this.refreshOrderIfNeeded();
        const data = this.currentData;
        const plot = data.plots[plotIndex];
        const at = nowSeconds();
        if (!plot || !plot.seedId || !isPlotMature(plot, at)) {
            this.reply(false, "还不能收获");
            return;
        }
        const seedId = plot.seedId as SeedId;
        const mutationId = plot.mutationId as MutationId;
        data.gold += cropValue(seedId, mutationId, this._order);
        this.addCollection(data, seedId, mutationId);
        plot.seedId = "";
        plot.plantedAt = 0;
        plot.mutationId = "none";
        plot.stolenCooldownUntil = 0;
        data.save(true);
        this.reply(true, "收获成功");
    }

    public net_activateShield(): void {
        const data = this.currentData;
        data.defenses.shieldUntil = nowSeconds() + SHIELD_SECONDS;
        data.save(true);
        this.reply(true, "护盾已开启");
    }

    public net_buyTrap(): void {
        const data = this.currentData;
        if (data.gold < 80) {
            this.reply(false, "金币不足");
            return;
        }
        data.gold -= 80;
        data.defenses.traps += 1;
        data.save(true);
        this.reply(true, "购买陷阱成功");
    }

    public net_steal(targetPlayerId: number, plotIndex: number): void {
        this.refreshOrderIfNeeded();
        const thief = this.currentPlayer;
        const target = Player.getPlayer(targetPlayerId);
        if (!target || target.playerId === thief.playerId) {
            this.reply(false, "不能偷这个目标");
            return;
        }
        const thiefData = this.currentData;
        const targetData = this.getPlayerData(target);
        const plot = targetData.plots[plotIndex];
        const at = nowSeconds();
        if (targetData.defenses.shieldUntil > at) {
            this.reply(false, "对方有护盾");
            return;
        }
        if (!plot || !plot.seedId || !isPlotMature(plot, at) || plot.stolenCooldownUntil > at) {
            this.reply(false, "目标暂时偷不了");
            return;
        }
        if (targetData.defenses.traps > 0) {
            targetData.defenses.traps -= 1;
            targetData.gold += 20;
            targetData.save(true);
            this.reply(false, "踩到陷阱了");
            this.getClient(target).net_receiveSnapshot(this.snapshotFor(target));
            return;
        }
        const reward = stealReward(plot.seedId as SeedId, plot.mutationId as MutationId, this._order);
        thiefData.gold += reward;
        plot.stolenCooldownUntil = at + STEAL_COOLDOWN_SECONDS;
        thiefData.save(true);
        targetData.save(true);
        this.reply(true, "偷到 " + reward + " 金币");
        this.getClient(target).net_receiveSnapshot(this.snapshotFor(target));
    }

    private reply(ok: boolean, message: string): void {
        this.getClient(this.currentPlayer).net_actionResult(ok, message, this.snapshotFor(this.currentPlayer));
    }

    private snapshotFor(player: Player): FarmSnapshot {
        const data = this.getPlayerData(player);
        return {
            gold: data.gold,
            baseLevel: data.baseLevel,
            plots: data.plots,
            defenses: data.defenses,
            order: this._order,
            collection: data.collection,
        };
    }

    private addCollection(data: PlayerFarmData, seedId: SeedId, mutationId: MutationId): void {
        const key = seedId + ":" + mutationId;
        if (data.collection.indexOf(key) < 0) data.collection.push(key);
    }

    private refreshOrderIfNeeded(): void {
        const at = nowSeconds();
        if (this._order.expiresAt > at) return;
        const seed = SEEDS[at % SEEDS.length];
        this._order = {
            seedId: seed.id,
            mutationId: "none",
            multiplier: 2,
            expiresAt: at + ORDER_SECONDS,
        };
    }
}
```

```ts
export class FarmInteractionClient {

        InputUtil.onKeyDown(Keys.Q, () => this.selectPrevPlot());
        InputUtil.onKeyDown(Keys.E, () => this.selectNextPlot());
        InputUtil.onKeyDown(Keys.Z, () => this._module.activateShield());
        InputUtil.onKeyDown(Keys.X, () => this._module.buyTrap());

    private selectPrevPlot(): void {
        this._selectedPlot = Math.max(0, this._selectedPlot - 1);

        this._selectedPlot = Math.min(11, this._selectedPlot + 1);

    }
```

```ts
import { BASE_LOCATIONS, BUILTIN_ASSETS, INITIAL_PLOTS } from "./NightMarketConfig";

export class FarmSceneBuilder {
    public async build(): Promise<void> {
        if (!SystemUtil.isServer()) return;
        for (let i = 0; i < BASE_LOCATIONS.length; i++) {
            await this.spawnBase(i, BASE_LOCATIONS[i]);
        }
    }

    private async spawnBase(index: number, origin: Vector): Promise<void> {
        const base = await GameObject.asyncSpawn(BUILTIN_ASSETS.cube) as Model;
        base.name = "FarmBase_" + index;
        base.worldTransform.position = origin;
        base.worldTransform.scale = new Vector(9, 9, 0.3);
        base.setCollision(CollisionStatus.On);

        for (let i = 0; i < INITIAL_PLOTS; i++) {
            const plot = await GameObject.asyncSpawn(BUILTIN_ASSETS.cube) as Model;
            plot.name = "FarmPlot_" + index + "_" + i;
            const x = (i % 2) * 240 - 120;
            const y = Math.floor(i / 2) * 240 - 120;
            plot.worldTransform.position = origin.clone().add(new Vector(x, y, 70));
            plot.worldTransform.scale = new Vector(1.8, 1.8, 0.15);
            plot.setCollision(CollisionStatus.On);
        }
    }
}
```
